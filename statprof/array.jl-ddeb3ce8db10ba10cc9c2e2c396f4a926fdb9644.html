55 """
56 CUDA.unsafe_free!(a::CuArray, [stream::CuStream])
57
58 Release the memory of an array for reuse by future allocations. This function is
59 automatically called by the finalizer when an array goes out of scope, but can be called
60 earlier to reduce pressure on the memory allocator.
61
62 By default, the operation is performed on the task-local stream. During task or process
63 finalization however, that stream may be destroyed already, so be sure to specify a safe
64 stream (i.e. `default_stream()`, which will ensure the operation will block on other
65 streams) when calling this function from a finalizer. For simplicity, the `unsafe_finalize!`
66 function does exactly that.
67 """
68 function unsafe_free!(xs::CuArray, stream::CuStream=stream())
69 # this call should only have an effect once, because both the user and the GC can call it
70 if xs.storage === nothing
71 return
72 elseif xs.storage.refcount[] < 0
73 throw(ArgumentError("Cannot free an unmanaged buffer."))
74 end
75
76 refcount = Threads.atomic_add!(xs.storage.refcount, -1)
77 if refcount == 1
78 context!(context(xs); skip_destroyed=true) do
79 free(xs.storage.buffer; stream)
80 end
81 end
82
83 # this array object is now dead, so replace its storage by a dummy one
84 xs.storage = nothing
85
86 return
87 end

89 function unsafe_finalize!(xs::CuArray)
90 # during task or process finalization, the local stream might be destroyed already, so
91 # use the default stream. additionally, since we don't use per-thread APIs, this default
92 # stream follows legacy semantics and will synchronize all other streams. this protects
93 # against freeing resources that are still in use.
94 #
95 # TODO: although this is still an asynchronous operation, even when using the default
96 # stream, it synchronizes "too much". we could do better, e.g., by keeping track of all
97 # streams involved, or by refcounting uses and decrementing that refcount after the
98 # operation using `cuLaunchHostFunc`. See CUDA.jl#778 and CUDA.jl#780 for details.
99 unsafe_free!(xs, default_stream())
100 # NOTE: we don't switch contexts here, but in unsafe_free!, as arrays are refcounted
101 # and we may not have to free the memory yet.
102 end

182
183 Wrap a `CuArray` object around the data at the address given by `ptr`. The pointer
184 element type `T` determines the array element type. `dims` is either an integer (for a 1d
185 array) or a tuple of the array dimensions. `own` optionally specified whether Julia should
186 take ownership of the memory, calling `cudaFree` when the array is no longer referenced. The
187 `ctx` argument determines the CUDA context where the data is allocated in.
188 """
189 function Base.unsafe_wrap(::Union{Type{CuArray},Type{CuArray{T}},Type{CuArray{T,N}},Type{CuArray{T,N,B}}},
190 ptr::CuPtr{T}, dims::NTuple{N,Int};
191 own::Bool=false, ctx::CuContext=context()) where {T,N,B}
192 Base.isbitstype(T) || error("Can only unsafe_wrap a pointer to a bits type")
193 sz = prod(dims) * sizeof(T)
194
195 # identify the buffer
196 buf = try
197 typ = memory_type(ptr)
198 if is_managed(ptr)
199 Mem.UnifiedBuffer(ctx, ptr, sz)
200 elseif typ == CU_MEMORYTYPE_DEVICE
201 # TODO: can we identify whether this pointer was allocated asynchronously?
202 Mem.DeviceBuffer(ctx, ptr, sz, false)
203 elseif typ == CU_MEMORYTYPE_HOST
204 Mem.HostBuffer(ctx, host_pointer(ptr), sz)
205 else
206 error("Unknown memory type; please file an issue.")
207 end
208 catch err
209 error("Could not identify the buffer type; are you passing a valid CUDA pointer to unsafe_wrap?")
210 end
211
212 if @isdefined(B) && typeof(buf) !== B
213 error("Declared buffer type does not match inferred buffer type.")
214 end

549 """
550 cu(A; unified=false)
551
552 Opinionated GPU array adaptor, which may alter the element type `T` of arrays:
553 * For `T<:AbstractFloat`, it makes a `CuArray{Float32}` for performance reasons.
554 (Except that `Float16` and `BFloat16` element types are not changed.)
555 * For `T<:Complex{<:AbstractFloat}` it makes a `CuArray{ComplexF32}`.
556 * For other `isbitstype(T)`, it makes a `CuArray{T}`.
557
558 By contrast, `CuArray(A)` never changes the element type.
559
560 Uses Adapt.jl to act inside some wrapper structs.
561
562 # Examples
563
564 ```
565 julia> cu(ones(3)')
566 1×3 adjoint(::CuArray{Float32, 1, CUDA.Mem.DeviceBuffer}) with eltype Float32:
567 1.0 1.0 1.0
568
569 julia> cu(zeros(1, 3); unified=true)
570 1×3 CuArray{Float32, 2, CUDA.Mem.UnifiedBuffer}:
571 0.0 0.0 0.0
572
573 julia> cu(1:3)
574 1:3
575
576 julia> CuArray(ones(3)') # ignores Adjoint, preserves Float64
577 1×3 CuArray{Float64, 2, CUDA.Mem.DeviceBuffer}:
578 1.0 1.0 1.0
579
580 julia> adapt(CuArray, ones(3)') # this restores Adjoint wrapper
581 1×3 adjoint(::CuArray{Float64, 1, CUDA.Mem.DeviceBuffer}) with eltype Float64:
582 1.0 1.0 1.0
583
584 julia> CuArray(1:3)
585 3-element CuArray{Int64, 1, CUDA.Mem.DeviceBuffer}:
586 1
587 2
588 3
589 ```
590 """
591 @inline cu(xs; unified::Bool=false) = adapt(CuArrayAdaptor{unified ? Mem.UnifiedBuffer : Mem.DeviceBuffer}(), xs)
592
593 Base.getindex(::typeof(cu), xs...) = CuArray([xs...])

722 # create a derived array (reinterpreted or reshaped) that's still a CuArray
723 @inline function _derived_array(::Type{T}, N::Int, a::CuArray, osize::Dims) where {T}
724 refcount = a.storage.refcount[]
725 @assert refcount != 0
726 if refcount > 0
727 Threads.atomic_add!(a.storage.refcount, 1)
728 end
729
730 offset = (a.offset * Base.elsize(a)) ÷ sizeof(T)
731
732 b = CuArray{T,N}(a.storage, osize; a.maxsize, offset)
733 if refcount > 0
734 finalizer(unsafe_finalize!, b)
735 end
736 return b
737 end

740 ## reinterpret
741
742 function Base.reinterpret(::Type{T}, a::CuArray{S,N}) where {T,S,N}
743 err = _reinterpret_exception(T, a)
744 err === nothing || throw(err)
745
746 if sizeof(T) == sizeof(S) # for N == 0
747 osize = size(a)
748 else
749 isize = size(a)
750 size1 = div(isize[1]*sizeof(S), sizeof(T))
751 osize = tuple(size1, Base.tail(isize)...)
752 end
753
754 return _derived_array(T, N, a, osize)
755 end
756
757 function _reinterpret_exception(::Type{T}, a::AbstractArray{S,N}) where {T,S,N}
758 if !isbitstype(T) || !isbitstype(S)
759 return _CuReinterpretBitsTypeError{T,typeof(a)}()
760 end
761 if N == 0 && sizeof(T) != sizeof(S)
762 return _CuReinterpretZeroDimError{T,typeof(a)}()
763 end
764 if N != 0 && sizeof(S) != sizeof(T)
765 ax1 = axes(a)[1]
766 dim = length(ax1)
767 if Base.rem(dim*sizeof(S),sizeof(T)) != 0
768 return _CuReinterpretDivisibilityError{T,typeof(a)}(dim)
769 end
770 if first(ax1) != 1
771 return _CuReinterpretFirstIndexError{T,typeof(a),typeof(ax1)}(ax1)
772 end
773 end
774 return nothing
775 end

806 ## reinterpret(reshape)
807
808 function Base.reinterpret(::typeof(reshape), ::Type{T}, a::CuArray) where {T}
809 N, osize = _base_check_reshape_reinterpret(T, a)
810 return _derived_array(T, N, a, osize)
811 end
812
813 # taken from reinterpretarray.jl
814 # TODO: move these Base definitions out of the ReinterpretArray struct for reuse
815 function _base_check_reshape_reinterpret(::Type{T}, a::CuArray{S}) where {T,S}
816 isbitstype(T) || throwbits(S, T, T)
817 isbitstype(S) || throwbits(S, T, S)
818 if sizeof(S) == sizeof(T)
819 N = ndims(a)
820 osize = size(a)
821 elseif sizeof(S) > sizeof(T)
822 d, r = divrem(sizeof(S), sizeof(T))
823 r == 0 || throwintmult(S, T)
824 N = ndims(a) + 1
825 osize = (d, size(a)...)
826 else
827 d, r = divrem(sizeof(T), sizeof(S))
828 r == 0 || throwintmult(S, T)
829 N = ndims(a) - 1
830 N > -1 || throwsize0(S, T, "larger")
831 axes(a, 1) == Base.OneTo(sizeof(T) ÷ sizeof(S)) || throwsize1(a, T)
832 osize = size(a)[2:end]
833 end
834 return N, osize
835 end

837 @noinline function throwbits(S::Type, T::Type, U::Type)
838 throw(ArgumentError("cannot reinterpret `$(S)` as `$(T)`, type `$(U)` is not a bits type"))
839 end

841 @noinline function throwintmult(S::Type, T::Type)
842 throw(ArgumentError("`reinterpret(reshape, T, a)` requires that one of `sizeof(T)` (got $(sizeof(T))) and `sizeof(eltype(a))` (got $(sizeof(S))) be an integer multiple of the other"))
843 end

845 @noinline function throwsize0(S::Type, T::Type, msg)
846 throw(ArgumentError("cannot reinterpret a zero-dimensional `$(S)` array to `$(T)` which is of a $msg size"))
847 end

849 @noinline function throwsize1(a::AbstractArray, T::Type)
850 throw(ArgumentError("`reinterpret(reshape, $T, a)` where `eltype(a)` is $(eltype(a)) requires that `axes(a, 1)` (got $(axes(a, 1))) be equal to 1:$(sizeof(T) ÷ sizeof(eltype(a))) (from the ratio of element sizes)"))
851 end

858
859 Resize `a` to contain `n` elements. If `n` is smaller than the current collection length,
860 the first `n` elements will be retained. If `n` is larger, the new elements are not

863 Note that this operation is only supported on managed buffers, i.e., not on arrays that are
864 created by `unsafe_wrap` with `own=false`.
865 """
866 function Base.resize!(A::CuVector{T}, n::Integer) where T
867 # TODO: add additional space to allow for quicker resizing
868 maxsize = n * sizeof(T)
869 bufsize = if Base.isbitsunion(T)
870 # type tag array past the data
871 maxsize + n
872 else
873 maxsize
874 end
875
876 new_storage = context!(context(A)) do
877 buf = alloc(typeof(A.storage.buffer), bufsize)
878 ptr = convert(CuPtr{T}, buf)
879 m = min(length(A), n)
880 if m > 0
881 unsafe_copyto!(ptr, pointer(A), m)
882 end
883 ArrayStorage(buf, 1)
884 end

886 unsafe_free!(A)
887 A.storage = new_storage
888 A.dims = (n,)
889 A.maxsize = maxsize
890 A.offset = 0
