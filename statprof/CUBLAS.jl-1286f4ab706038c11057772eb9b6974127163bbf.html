211 function _log_message(blob)
212 # the message format isn't documented, but it looks like a message starts with a capital
213 # and the severity (e.g. `I!`), and subsequent lines start with a lowercase mark (`!i`)
214 #
215 # lines are separated by a \0 if they came in separately, but there may also be multiple
216 # actual lines separated by \n in each message.
217 for message in split(blob, r"[\0\n]+(?=[A-Z]!)")
218 code = message[1]
219 lines = split(message[3:end], r"[\0\n]+[a-z]!")
220 submessage = join(lines, '\n')
221 if code == 'I'
222 @debug submessage
223 elseif code == 'W'
224 @warn submessage
225 elseif code == 'E'
226 @error submessage
227 elseif code == 'F'
228 error(submessage)
229 else
230 @info "Unknown log message, please file an issue.\n$message"
231 end
232 end
233 return
234 end
