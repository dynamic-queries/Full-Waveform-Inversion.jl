105 function pool_cleanup()
106 idle_counters = Base.fill(0, ndevices())
107 while true
108 for (i, dev) in enumerate(devices())
109 stream_ordered(dev) || continue
110
111 status = pool_status(dev)
112 status[] === nothing && continue
113
114 if status[]
115 idle_counters[i] = 0
116 else
117 idle_counters[i] += 1
118 end

122 # the pool hasn't been used for a while, so reclaim unused buffers
