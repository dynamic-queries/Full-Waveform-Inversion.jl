49 function OperatorConv(ch::Pair{S, S},
50 modes::NTuple{N, S},
51 Transform::Type{<:AbstractTransform};
52 init = (dims...) -> Flux.glorot_uniform(eltype(Transform), dims...),
53 permuted = false,
54 T::DataType = eltype(Transform)) where {S <: Integer, N}
55 in_chs, out_chs = ch
56 scale = one(T) / (in_chs * out_chs)
57 weights = scale * init(prod(modes), in_chs, out_chs)
58 transform = Transform(modes)
59
60 return OperatorConv{permuted}(weights, in_chs, out_chs, transform)

122 """
123 OperatorKernel(ch, modes, σ=identity; permuted=false)

125 ## Arguments

127 * `ch`: A `Pair` of input and output channel size for spectral convolution `in_ch=>out_ch`,

154 function OperatorKernel(ch::Pair{S, S},
155 modes::NTuple{N, S},
156 Transform::Type{<:AbstractTransform},
157 σ = identity;
158 permuted = false) where {S <: Integer, N}
159 linear = permuted ? Conv(Tuple(ones(Int, length(modes))), ch) :
160 Dense(ch.first, ch.second)
161 conv = OperatorConv(ch, modes, Transform; permuted = permuted)
162
163 return OperatorKernel(linear, conv, σ)
