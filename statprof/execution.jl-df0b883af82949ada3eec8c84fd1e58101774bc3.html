8 """
9 @cuda [kwargs...] func(args...)
10
11 High-level interface for executing code on a GPU. The `@cuda` macro should prefix a call,
12 with `func` a callable function or object that should return nothing. It will be compiled to
13 a CUDA function upon first use, and to a certain extent arguments will be converted and
14 managed automatically using `cudaconvert`. Finally, a call to `cudacall` is
15 performed, scheduling a kernel launch on the current CUDA context.
16
17 Several keyword arguments are supported that influence the behavior of `@cuda`.
18 - `launch`: whether to launch this kernel, defaults to `true`. If `false` the returned
19 kernel object should be launched by calling it and passing arguments again.
20 - `dynamic`: use dynamic parallelism to launch device-side kernels, defaults to `false`.
21 - arguments that influence kernel compilation: see [`cufunction`](@ref) and
22 [`dynamic_cufunction`](@ref)
23 - arguments that influence kernel launch: see [`CUDA.HostKernel`](@ref) and
24 [`CUDA.DeviceKernel`](@ref)

177 @inline @generated function call(kernel::AbstractKernel{F,TT}, args...; call_kwargs...) where {F,TT}
178 sig = Tuple{F, TT.parameters...} # Base.signature_type with a function type
179 args = (:(kernel.f), (:( args[$i] ) for i in 1:length(args))...)
180
181 # filter out arguments that shouldn't be passed
182 predicate = dt -> isghosttype(dt) || Core.Compiler.isconstType(dt)
183 to_pass = map(!predicate, sig.parameters)
184 call_t = Type[x[1] for x in zip(sig.parameters, to_pass) if x[2]]
185 call_args = Union{Expr,Symbol}[x[1] for x in zip(args, to_pass) if x[2]]
186
187 # replace non-isbits arguments (they should be unused, or compilation would have failed)
188 # alternatively, make it possible to `launch` with non-isbits arguments.
189 for (i,dt) in enumerate(call_t)
190 if !isbitstype(dt)
191 call_t[i] = Ptr{Any}
192 call_args[i] = :C_NULL
193 end
194 end
195
196 # add the kernel state
197 pushfirst!(call_t, KernelState)
198 pushfirst!(call_args, :(kernel.state))
199
200 # finalize types
201 call_tt = Base.to_tuple_type(call_t)
202
203 quote
204 cudacall(kernel.fun, $call_tt, $(call_args...); call_kwargs...)
205 end
206 end

350 @timeit_ci "compile" function cufunction_compile(@nospecialize(job::CompilerJob), ctx)
351 # lower to PTX
352 mi, mi_meta = @timeit_ci "emit_julia" GPUCompiler.emit_julia(job)
353 ir, ir_meta = @timeit_ci "emit_llvm" GPUCompiler.emit_llvm(job, mi; ctx)
354 asm, asm_meta = @timeit_ci "emit_asm" GPUCompiler.emit_asm(job, ir; format=LLVM.API.LLVMAssemblyFile)
355
356 # remove extraneous debug info on lower debug levels
357 if Base.JLOptions().debug_level < 2
358 # LLVM sets `.target debug` as soon as the debug emission kind isn't NoDebug. this
359 # is unwanted, as the flag makes `ptxas` behave as if `--device-debug` were set.
360 # ideally, we'd need something like LocTrackingOnly/EmitDebugInfo from D4234, but
361 # that got removed in favor of NoDebug in D18808, seemingly breaking the use case of
362 # only emitting `.loc` instructions...
363 #
364 # according to NVIDIA, "it is fine for PTX producers to produce debug info but not
365 # set `.target debug` and if `--device-debug` isn't passed, PTXAS will compile in
366 # release mode".
367 asm = replace(asm, r"(\.target .+), debug" => s"\1")
368 end
369
370 # check if we'll need the device runtime
371 undefined_fs = filter(collect(functions(ir))) do f
372 isdeclaration(f) && !LLVM.isintrinsic(f)
373 end
374 intrinsic_fns = ["vprintf", "malloc", "free", "__assertfail",
375 "__nvvm_reflect" #= TODO: should have been optimized away =#]
376 needs_cudadevrt = !isempty(setdiff(LLVM.name.(undefined_fs), intrinsic_fns))
377
378 # find externally-initialized global variables; we'll access those using CUDA APIs.
379 external_gvars = filter(isextinit, collect(globals(ir))) .|> LLVM.name
380
381 # prepare invocations of CUDA compiler tools
382 ptxas_opts = String[]
383 nvlink_opts = String[]
384 ## debug flags
385 if Base.JLOptions().debug_level == 1
386 push!(ptxas_opts, "--generate-line-info")
387 elseif Base.JLOptions().debug_level >= 2
388 push!(ptxas_opts, "--device-debug")
389 push!(nvlink_opts, "--debug")
390 end
391 ## relocatable device code
392 if needs_cudadevrt
393 push!(ptxas_opts, "--compile-only")
394 end
395
396 arch = "sm_$(job.target.cap.major)$(job.target.cap.minor)"
397
398 # compile to machine code
399 # NOTE: we use tempname since mktemp doesn't support suffixes, and mktempdir is slow
400 ptx_input = tempname(cleanup=false) * ".ptx"
401 ptxas_output = tempname(cleanup=false) * ".cubin"
402 write(ptx_input, asm)
403
404 # we could use the driver's embedded JIT compiler, but that has several disadvantages:
405 # 1. fixes and improvements are slower to arrive, by using `ptxas` we only need to
406 # upgrade the toolkit to get a newer compiler;
407 # 2. version checking is simpler, we otherwise need to use NVML to query the driver
408 # version, which is hard to correlate to PTX JIT improvements;
409 # 3. if we want to be able to use newer (minor upgrades) of the CUDA toolkit on an
410 # older driver, we should use the newer compiler to ensure compatibility.
411 append!(ptxas_opts, [
412 "--verbose",
413 "--gpu-name", arch,
414 "--output-file", ptxas_output,
415 ptx_input
416 ])
417 proc, log = @timeit_ci "ptxas" run_and_collect(`$(ptxas()) $ptxas_opts`)
418 log = strip(log)
419 if !success(proc)
420 reason = proc.termsignal > 0 ? "ptxas received signal $(proc.termsignal)" :
421 "ptxas exited with code $(proc.exitcode)"
422 msg = "Failed to compile PTX code ($reason)"
423 if !isempty(log)
424 msg *= "\n" * log
425 end
426 msg *= "\nIf you think this is a bug, please file an issue and attach $(ptx_input)"
427 error(msg)
428 elseif !isempty(log)
429 @debug "PTX compiler log:\n" * log
430 end
431 rm(ptx_input)
432
433 # link device libraries, if necessary
434 #
435 # this requires relocatable device code, which prevents certain optimizations and
436 # hurts performance. as such, we only do so when absolutely necessary.
437 # TODO: try LTO, `--link-time-opt --nvvmpath /opt/cuda/nvvm`.
438 # fails with `Ignoring -lto option because no LTO objects found`

441 append!(nvlink_opts, [
442 "--verbose", "--extra-warnings",
443 "--arch", arch,
444 "--library-path", dirname(libcudadevrt()),
445 "--library", "cudadevrt",
446 "--output-file", nvlink_output,

448 ])
