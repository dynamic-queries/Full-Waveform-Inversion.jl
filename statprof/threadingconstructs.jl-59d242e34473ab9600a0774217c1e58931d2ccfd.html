25 function threading_run(fun, static)
26 ccall(:jl_enter_threaded_region, Cvoid, ())
27 n = nthreads()
28 tasks = Vector{Task}(undef, n)
29 for i = 1:n
30 t = Task(() -> fun(i)) # pass in tid
31 t.sticky = static
32 static && ccall(:jl_set_task_tid, Cint, (Any, Cint), t, i-1)
33 tasks[i] = t
34 schedule(t)
35 end
36 try
37 for i = 1:n
38 wait(tasks[i])
39 end
40 finally
41 ccall(:jl_exit_threaded_region, Cvoid, ())
42 end
43 end

45 function _threadsfor(iter, lbody, schedule)
46 lidx = iter.args[1] # index
47 range = iter.args[2]
48 quote
49 local threadsfor_fun
50 let range = $(esc(range))
51 function threadsfor_fun(tid=1; onethread=false)
52 r = range # Load into local variable
53 lenr = length(r)
54 # divide loop iterations among threads
55 if onethread
56 tid = 1
57 len, rem = lenr, 0
58 else
59 len, rem = divrem(lenr, nthreads())
60 end
61 # not enough iterations for all the threads?
62 if len == 0
63 if tid > rem
64 return
65 end
66 len, rem = 1, 0
67 end
68 # compute this thread's iterations
69 f = firstindex(r) + ((tid-1) * len)
70 l = f + len - 1
71 # distribute remaining iterations evenly
72 if rem > 0
73 if tid <= rem
74 f = f + (tid-1)
75 l = l + tid
76 else
77 f = f + rem
78 l = l + rem
79 end
80 end
81 # run this thread's iterations
82 for i = f:l
83 local $(esc(lidx)) = @inbounds r[i]
84 $(esc(lbody))
85 end
86 end
87 end
88 if $(schedule === :dynamic || schedule === :default)
89 threading_run(threadsfor_fun, false)
90 elseif ccall(:jl_in_threaded_region, Cint, ()) != 0 # :static
91 error("`@threads :static` cannot be used concurrently or nested")
92 else # :static
93 threading_run(threadsfor_fun, true)
94 end
95 nothing
96 end
97 end

99 """
100 Threads.@threads [schedule] for ... end
101
102 A macro to execute a `for` loop in parallel. The iteration space is distributed to
103 coarse-grained tasks. This policy can be specified by the `schedule` argument. The
104 execution of the loop waits for the evaluation of all iterations.
105
106 See also: [`@spawn`](@ref Threads.@spawn) and
107 `pmap` in [`Distributed`](@ref man-distributed).
108
109 # Extended help
110
111 ## Semantics
112
113 Unless stronger guarantees are specified by the scheduling option, the loop executed by
114 `@threads` macro have the following semantics.
115
116 The `@threads` macro executes the loop body in an unspecified order and potentially
117 concurrently. It does not specify the exact assignments of the tasks and the worker threads.
118 The assignments can be different for each execution. The loop body code (including any code
119 transitively called from it) must not make any assumptions about the distribution of
120 iterations to tasks or the worker thread in which they are executed. The loop body for each
121 iteration must be able to make forward progress independent of other iterations and be free
122 from data races. As such, invalid synchronizations across iterations may deadlock while
123 unsynchronized memory accesses may result in undefined behavior.
124
125 For example, the above conditions imply that:
126
127 - The lock taken in an iteration *must* be released within the same iteration.
128 - Communicating between iterations using blocking primitives like `Channel`s is incorrect.
129 - Write only to locations not shared across iterations (unless a lock or atomic operation is
130 used).
131 - The value of [`threadid()`](@ref Threads.threadid) may change even within a single
132 iteration.
133
134 ## Schedulers
135
136 Without the scheduler argument, the exact scheduling is unspecified and varies across Julia
137 releases. Currently, `:dynamic` is used when the scheduler is not specified.
138
139 !!! compat "Julia 1.5"
140 The `schedule` argument is available as of Julia 1.5.
141
142 ### `:dynamic` (default)
143
144 `:dynamic` scheduler executes iterations dynamically to available worker threads. Current
145 implementation assumes that the workload for each iteration is uniform. However, this
146 assumption may be removed in the future.
147
148 This scheduling option is merely a hint to the underlying execution mechanism. However, a
149 few properties can be expected. The number of `Task`s used by `:dynamic` scheduler is
150 bounded by a small constant multiple of the number of available worker threads
151 ([`nthreads()`](@ref Threads.nthreads)). Each task processes contiguous regions of the
152 iteration space. Thus, `@threads :dynamic for x in xs; f(x); end` is typically more
153 efficient than `@sync for x in xs; @spawn f(x); end` if `length(xs)` is significantly
154 larger than the number of the worker threads and the run-time of `f(x)` is relatively
155 smaller than the cost of spawning and synchronizaing a task (typically less than 10
156 microseconds).
157
158 !!! compat "Julia 1.8"
159 The `:dynamic` option for the `schedule` argument is available and the default as of Julia 1.8.
160
161 ### `:static`
162
163 `:static` scheduler creates one task per thread and divides the iterations equally among
164 them, assigning each task specifically to each thread. In particular, the value of
165 [`threadid()`](@ref Threads.threadid) is guranteed to be constant within one iteration.
166 Specifying `:static` is an error if used from inside another `@threads` loop or from a
167 thread other than 1.
168
169 !!! note
170 `:static` scheduling exists for supporting transition of code written before Julia 1.3.
171 In newly written library functions, `:static` scheduling is discouraged because the
172 functions using this option cannot be called from arbitrary worker threads.
173
174 ## Example
175
176 To illustrate of the different scheduling strategies, consider the following function
177 `busywait` containing a non-yielding timed loop that runs for a given number of seconds.
178
179 ```julia-repl
180 julia> function busywait(seconds)
181 tstart = time_ns()
182 while (time_ns() - tstart) / 1e9 < seconds
183 end
184 end
185
186 julia> @time begin
187 Threads.@spawn busywait(5)
188 Threads.@threads :static for i in 1:Threads.nthreads()
189 busywait(1)
190 end
191 end
192 6.003001 seconds (16.33 k allocations: 899.255 KiB, 0.25% compilation time)
193
194 julia> @time begin
195 Threads.@spawn busywait(5)
196 Threads.@threads :dynamic for i in 1:Threads.nthreads()
197 busywait(1)
198 end
199 end
200 2.012056 seconds (16.05 k allocations: 883.919 KiB, 0.66% compilation time)
201 ```
202
203 The `:dynamic` example takes 2 seconds since one of the non-occupied threads is able
204 to run two of the 1-second iterations to complete the for loop.
205 """
206 macro threads(args...)
207 na = length(args)
208 if na == 2
209 sched, ex = args
210 if sched isa QuoteNode
211 sched = sched.value
212 elseif sched isa Symbol
213 # for now only allow quoted symbols
214 sched = nothing
215 end
216 if sched !== :static && sched !== :dynamic
217 throw(ArgumentError("unsupported schedule argument in @threads"))
218 end
219 elseif na == 1
220 sched = :default
221 ex = args[1]
222 else
223 throw(ArgumentError("wrong number of arguments in @threads"))
224 end
225 if !(isa(ex, Expr) && ex.head === :for)
226 throw(ArgumentError("@threads requires a `for` loop expression"))
227 end
228 if !(ex.args[1] isa Expr && ex.args[1].head === :(=))
229 throw(ArgumentError("nested outer loops are not currently supported by @threads"))
230 end
231 return _threadsfor(ex.args[1], ex.args[2], sched)
232 end

234 """
235 Threads.@spawn expr
236
237 Create a [`Task`](@ref) and [`schedule`](@ref) it to run on any available thread.
238 The task is allocated to a thread after it becomes available. To wait for the task
239 to finish, call [`wait`](@ref) on the result of this macro, or call [`fetch`](@ref) to
240 wait and then obtain its return value.
241
242 Values can be interpolated into `@spawn` via `\$`, which copies the value directly into the
243 constructed underlying closure. This allows you to insert the _value_ of a variable,
244 isolating the asynchronous code from changes to the variable's value in the current task.
245
246 !!! note
247 See the manual chapter on threading for important caveats.
248
249 !!! compat "Julia 1.3"
250 This macro is available as of Julia 1.3.
251
252 !!! compat "Julia 1.4"
253 Interpolating values via `\$` is available as of Julia 1.4.
254 """
